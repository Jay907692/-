local function ValidateKey(inputKey)
    local cleanedKey = inputKey:gsub("%s+", ""):upper()
    for _, validKey in ipairs(ValidKeys) do
        if cleanedKey == validKey then
            return true
        end
    end
    return false
end

local dragging = false
local dragStart = Vector2.new()
local originalPos = MainPanel.Position

MainPanel.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position - originalPos.Position
        MainPanel.Active = true
    end
end)

MainPanel.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement and dragging then
        originalPos = UDim2.new(0, input.Position.X - dragStart.X, 0, input.Position.Y - dragStart.Y)
        MainPanel.Position = originalPos
    end
end)

MainPanel.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
        MainPanel.Active = false
    end
end)
